name: "Nawh Clean Production Build"

on:
  push:
    branches:
      - main
  workflow_dispatch:

jobs:
  build:
    runs-on: ubuntu-latest
    permissions:
      contents: write
    env:
      FORCE_JAVASCRIPT_ACTIONS_TO_NODE24: true

    steps:
      # 1. سحب الكود من المستودع
      - name: 1. Checkout Code
        uses: actions/checkout@v4

      # 2. تهيئة بيئة جافا المتوافقة مع أندرويد
      - name: 2. Setup Java 17
        uses: actions/setup-java@v4
        with:
          distribution: 'temurin'
          java-version: '17'

      # 3. تهيئة بيئة نود
      - name: 3. Setup Node.js 24
        uses: actions/setup-node@v4
        with:
          node-version: '24'

      # 4. تنظيف المشروع وتثبيت الاعتماديات
      - name: 4. Deep Clean & Install
        run: |
          rm -rf node_modules package-lock.json android ios dist_web
          npm install --legacy-peer-deps
          # تثبيت الإضافات الضرورية لضمان وجودها
          npm install onesignal-cordova-plugin @capacitor/push-notifications @capacitor/local-notifications --save --legacy-peer-deps
          npm run build

      # 5. تهيئة منصة أندرويد من الصفر
      - name: 5. Initialize Android (Fresh Start)
        run: |
          npx cap add android
          npx cap sync android

      # 6. حقن ملفات الهوية والميزات (Firebase, App Identity, Start.io)
      - name: 6. Inject Firebase, App Identity & Ad Network (Start.io)
        shell: bash
        run: |
          # حقن ملف جوجل وتعديل الـ package_name إلى المعرّف الخاص بنا
          cat <<EOF > android/app/google-services.json
          {
            "project_info": {
              "project_number": "162488255991",
              "project_id": "raqqa-43dc8",
              "storage_bucket": "raqqa-43dc8.firebasestorage.app"
            },
            "client": [
              {
                "client_info": {
                  "mobilesdk_app_id": "1:162488255991:android:73d6299f11a1b7aec61af2",
                  "android_client_info": { "package_name": "ai.nawah.app" }
                },
                "api_key": [ { "current_key": "AIzaSyAKjsgnoHnGGr3urhm6Kpu7RvxN2dp6sJQ" } ],
                "services": { "appinvite_service": { "other_platform_oauth_client": [] } }
              }
            ],
            "configuration_version": "1"
          }
          EOF

          # ملفات الإعدادات
          STRINGS="android/app/src/main/res/values/strings.xml"
          APP_GRADLE="android/app/build.gradle"
          MANIFEST="android/app/src/main/AndroidManifest.xml"
          
          # تعديل اسم التطبيق ليصبح (نواة AI) والحزمة لـ (ai.nawah.app)
          sed -i 's/<string name="app_name">.*<\/string>/<string name="app_name">نواة AI<\/string>/' $STRINGS
          sed -i 's/namespace ".*"/namespace "ai.nawah.app"/' $APP_GRADLE
          sed -i 's/applicationId ".*"/applicationId "ai.nawah.app"/' $APP_GRADLE
          
          # إضافة الأذونات اللازمة (الإنترنت، الإشعارات، التسجيل)
          sed -i '/<application/i \    <uses-permission android:name="android.permission.INTERNET" />\n    <uses-permission android:name="android.permission.ACCESS_NETWORK_STATE" />\n    <uses-permission android:name="android.permission.RECORD_AUDIO" />\n    <uses-permission android:name="android.permission.POST_NOTIFICATIONS" />' $MANIFEST

          # إضافة معرفات Start.io داخل وسم <application> في المانيفست
          sed -i '/<application[^>]*>/a \        <meta-data android:name="com.startapp.sdk.APPLICATION_ID" android:value="203477356" />' $MANIFEST

      # 7. توليد الأيقونات وصور البداية للتطبيق
      - name: 7. Generate App Icons & Splash
        run: |
          mkdir -p assets
          # البحث عن الصور ونسخها للأصول
          find . -name "icon-only.png" -exec cp {} assets/icon-only.png \;
          find . -name "icon-background.png" -exec cp {} assets/icon-background.png \;
          find . -name "icon-foreground.png" -exec cp {} assets/icon-foreground.png \;
          npx capacitor-assets generate --android

      # 8. بناء ملفات أندرويد الثنائية
      - name: 8. Build Android Binaries
        run: |
          cd android
          chmod +x gradlew
          ./gradlew clean
          ./gradlew assembleRelease
          ./gradlew bundleRelease

      # 9. توقيع ملفات الـ APK والـ AAB
      - name: 9. Sign APK & AAB
        run: |
          # توليد مفتاح توقيع مؤقت خاص بـ Nawh
          keytool -genkey -v -keystore nawh_release.jks -alias nawh_alias -keyalg RSA -keysize 2048 -validity 10000 \
            -storepass "nawh_pass_2026" -keypass "nawh_pass_2026" \
            -dname "CN=NawhApp, OU=Dev, O=Nawh, L=Cairo, S=Egypt, C=EG"
          
          # توقيع الـ APK
          APK_PATH=$(find android/app/build/outputs/apk/release/ -name "*.apk" | head -n 1)
          /usr/local/lib/android/sdk/build-tools/34.0.0/apksigner sign --ks nawh_release.jks \
            --ks-pass pass:"nawh_pass_2026" --out Nawh-Final-Signed.apk "$APK_PATH"
          
          # توقيع الـ AAB
          AAB_PATH=$(find android/app/build/outputs/bundle/release/ -name "*.aab" | head -n 1)
          jarsigner -keystore nawh_release.jks -storepass "nawh_pass_2026" "$AAB_PATH" nawh_alias
          cp "$AAB_PATH" Nawh-Final-Signed.aab

      # 10. نشر النسخة النهائية كـ Release على المستودع
      - name: 10. Publish Release
        uses: softprops/action-gh-release@v2
        with:
          tag_name: v${{ github.run_number }}
          name: "Nawh Production Build v${{ github.run_number }}"
          files: |
            Nawh-Final-Signed.apk
            Nawh-Final-Signed.aab
            nawh_release.jks
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
